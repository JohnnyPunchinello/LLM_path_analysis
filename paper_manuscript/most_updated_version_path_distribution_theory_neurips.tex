\begin{abstract}
We propose \emph{path distribution theory}, a framework for understanding emergent
abilities in large language models through the lens of computational graph topology.
By modelling transformer architectures as block-level directed acyclic graphs (DAGs),
we show that residual connections create an exponential family of distinct
input-to-output paths whose length distribution follows architecture-specific forms.
We prove that standard pre-LN transformers yield binomial path distributions with
$4^L$ paths for $L$ layers, while parallel-attention architectures (GPT-J) yield
trinomial distributions with $3^L$ paths, and cross-layer residual architectures
(DenseNet, KIMI-style skip-2) produce super-exponential path growth.

A central methodological contribution is the \emph{empirical path distribution}
$\emppathdist$, computed by applying Attribution Patching (AtP) to score each
architectural edge and restricting path counting to the resulting active subgraph
$G_{\text{active}}$. The gap $\gap = \entropy(\pathdist) - \entropy(\emppathdist)$,
which we term the \emph{Synergy Gap}, measures how much a trained model specialises
its computational paths for a given task.

We introduce four distribution-derived metrics---\emph{path entropy}, \emph{effective
depth spectrum}, \emph{tail-mass ratio}, and \emph{Synergy Gap}---and design
experiments to test whether these metrics predict the onset of emergent abilities
across model scales more reliably than parameter count or FLOP budget alone. We
hypothesise that emergence correlates with a critical threshold in path entropy
($\entropy(\pathdist) \approx 3.2$\,bits), and that the Synergy Gap increases
monotonically with task complexity, providing a complementary lens on what
computations a trained model actually activates.
\end{abstract}

\section{Introduction}
\label{sec:introduction}
% ============================================================================

The phenomenon of \emph{emergent abilities}---capabilities that appear abruptly as
language models scale---remains one of the most striking and least understood aspects
of modern AI \citep{wei2022emergent}. While scaling laws predict smooth performance
improvements in loss \citep{kaplan2020scaling, hoffmann2022training}, specific
downstream tasks exhibit sharp phase transitions at seemingly unpredictable model
sizes.

Existing explanations fall into three camps: (1) emergent abilities are an artefact
of evaluation metrics \citep{schaeffer2024emergent}; (2) they are driven by
grokking-like phase transitions in learned representations \citep{power2022grokking};
or (3) they arise from the compositional structure of the computation itself. We
pursue the third direction, arguing that the \emph{topology} of residual connections
creates a rich space of computational paths whose distribution determines what
computations a model can express.

Our key insight is simple: in a standard transformer, every residual connection
creates a binary branch point. A signal entering the model can either pass through a
computation block (attention or MLP) or skip it via the residual stream. With
$\residuals = 2L$ such branch points, there are $2^{2L} = 4^L$ distinct paths from
input to output. The \emph{distribution} of these path lengths---not just their
count---captures the effective computational diversity of the architecture.

Critically, the analytical path distribution describes what a model \emph{could}
compute. The mechanistic interpretability literature
\citep{elhage2021mathematical,conmy2023automated} shows that trained models selectively
activate subsets of their computation graph depending on the input. We therefore
introduce the \emph{empirical path distribution}, derived via Attribution Patching
(AtP) on a trained model, which captures what computations are \emph{actually} used.

\paragraph{Contributions.}
\begin{enumerate}
    \item We formalise \textbf{path distribution theory} for transformer architectures,
          proving exact distributional forms for 8 architectural families
          (Section~\ref{sec:theory}).
    \item We introduce \textbf{four predictive metrics}---path entropy $\entropy(\pathdist)$,
          effective depth spectrum width $\sigma_d$, tail-mass ratio $\tau_k$, and
          Synergy Gap $\gap$---that together characterise both the structural richness
          of an architecture and the task-specific specialisation of a trained model
          (Sections~\ref{sec:metrics} and~\ref{sec:empirical}).
    \item We introduce the \textbf{empirical path distribution} $\emppathdist$ via
          Attribution Patching, defining the active subgraph $G_{\text{active}}$ by
          thresholding AtP scores at a task-level quantile, and define the
          \textbf{Synergy Gap} $\gap = \entropy(\pathdist) - \entropy(\emppathdist)$
          as an interpretability diagnostic measuring task-driven path specialisation
          (Section~\ref{sec:empirical}).
    \item We design experiments to test the \textbf{emergence--entropy correlation}
          across 12 tasks and 8 architectural families, providing falsifiable predictions
          and infrastructure for evaluating the theory at scale
          (Section~\ref{sec:experiments}).
    \item We show that \textbf{architectural modifications to the path distribution}
          (adding cross-layer residuals) shift the effective entropy without changing
          parameter count, suggesting a new design principle for inducing emergence
          (Section~\ref{sec:design}).
\end{enumerate}

\section{Related Work}
\label{sec:related}
% ============================================================================

\paragraph{Emergent abilities and scaling laws.}
\citet{wei2022emergent} documented emergent abilities across diverse tasks, while
\citet{schaeffer2024emergent} argued these may be metric artefacts. \citet{kaplan2020scaling}
and \citet{hoffmann2022training} established power-law scaling of loss with compute, data,
and parameters, but these laws do not predict task-specific phase transitions. Our work
bridges this gap by identifying an architecture-level quantity (path entropy) that
predicts when smooth scaling transitions into emergent capability.

\paragraph{Residual networks as ensembles.}
\citet{veit2016residual} showed that ResNets behave as ensembles of shallow networks,
with most gradient signal flowing through short paths. \citet{huang2016deep} leveraged
this by stochastically dropping layers (stochastic depth). We extend this
``ensemble of subnetworks'' view to transformers, providing exact combinatorics for the
path distribution and connecting it to emergent capabilities.

\paragraph{Depth and compositionality.}
\citet{eldan2016power} proved depth separation results for neural networks, and
\citet{merrill2023expressive} characterised the expressive power of transformers as a
function of depth. The connection between depth and compositional generalisation was
studied by \citet{ontanon2022making}. Our path distribution framework unifies these
results: the effective depth of a model is not its layer count but the support of its
path distribution.

\paragraph{Mechanistic interpretability and activation patching.}
\citet{elhage2021mathematical} introduced the residual stream decomposition, framing
the transformer as a sum of paths through attention heads and MLP blocks---the direct
precursor to our block-level DAG. \citet{meng2022locating} demonstrated that activation
patching can localise factual associations to specific layers; Attribution Patching (AtP)
\citep{nanda2023attribution} provides a first-order linear approximation of this
intervention, enabling efficient attribution of every edge without a separate forward pass
per edge. \citet{conmy2023automated} (ACDC) extended this to automated circuit discovery,
finding minimal subgraphs ($G_{\text{active}}$ in our terminology) that are sufficient for
a task. Our empirical path distribution generalises this by quantifying the path-length
structure of $G_{\text{active}}$, connecting circuit-finding to the distributional metrics
of the analytical theory.

\paragraph{Architectural innovations in residual connections.}
DenseNet \citep{huang2017densely} introduced dense cross-layer connections. Mixture of
Depths \citep{raposo2024mixture} allows dynamic depth allocation. U-Net-style transformers
\citep{bao2023all} add symmetric skip connections. We provide the first unified analysis
of how each design affects the path distribution and, consequently, emergent behaviour.

\section{Path Distribution Theory}
\label{sec:theory}
% ============================================================================

\subsection{Block-Level Computational Graphs}

We model a transformer as a directed acyclic graph (DAG) $G = (V, E)$ at the
\emph{block level}, where each node $v \in V$ represents a residual-stream checkpoint
and edges represent computations.  We distinguish two edge types:

\begin{itemize}
    \item \textbf{Skip edges} (type $s$): identity connections (residual bypasses),
          carrying path-length delta $\delta = 0$.
    \item \textbf{Compute edges} (type $c$): edges passing through an attention or
          MLP block, carrying $\delta = 1$.
\end{itemize}

\begin{definition}[Computational Path]
A \emph{computational path} $p$ is a node sequence $(v_1, \ldots, v_k)$ with
$(v_i, v_{i+1}) \in E$ for all $i$, from the embedding node to the LM-head node.
The \emph{(block-level) length} of $p$ is $|p| = \sum_{i} \delta(e_i)$, the number
of compute edges traversed.
\end{definition}

\begin{definition}[Path Distribution]
The \emph{path distribution} $\pathdist$ assigns probability proportional to path
count to each block-level length $\ell$:
$$\pathdist(\ell) = \frac{|\{p \in \pathcount : |p| = \ell\}|}{|\pathcount|}$$
where $\pathcount$ is the set of all source-to-sink paths.
\end{definition}

\begin{remark}[Block-level vs.\ fine-grained representations]
A finer-grained representation (used in Table~\ref{tab:architectures}) counts each
norm, block, and residual-add as a separate edge, assigning skip connections $\delta = 1$
and compute branches $\delta = 3$ (norm\,$\to$\,block\,$\to$\,add) in a pre-LN model,
with a fixed backbone of 2 edges. This yields an affine transform
$\ell_{\text{fine}} = 2 + 2\ell_{\text{block}}$ for pre-LN sequential models, preserving
all distributional shape statistics (skewness, kurtosis, normalised entropy). Algorithm~1
operates in block-level units.
\end{remark}

\subsection{Exact Path Distributions for Transformer Families}

\begin{theorem}[Pre-LN / RMSNorm Sequential Path Distribution]
\label{thm:preln}
A pre-LN (or pre-RMSNorm) transformer (GPT-2, LLaMA) with $L$ layers has
$2L$ binary branch points and $|\pathcount| = 4^L$ total paths. In block-level units
the path distribution is binomial:
$$\pathdist(\ell) = \binom{2L}{\ell}\left(\tfrac{1}{2}\right)^{2L}, \quad \ell = 0, 1, \ldots, 2L$$
with mean $\mu = L$, variance $\sigma^2 = L/2$, and zero skewness.  In fine-grained
edge units the distribution is the same shape, with $\mu_{\text{fine}} = 4L + 2$.
\end{theorem}

\begin{proof}[Proof sketch]
Each layer contributes two independent Bernoulli$(\tfrac{1}{2})$ branch points
(one for attention, one for MLP), each incrementing the block-level path length by
0 (skip) or 1 (compute). The total block-level length is $\sum_{i=1}^{2L} X_i$ where
$X_i \sim \text{Bernoulli}(\tfrac{1}{2})$, yielding Binomial$(2L, \tfrac{1}{2})$.
\qed
\end{proof}

\begin{theorem}[GPT-J Parallel Path Distribution]
\label{thm:gptj}
GPT-J's parallel attention+MLP architecture with $L$ layers has $|\pathcount| = 3^L$
total paths. Each layer contributes a ternary branch (residual skip, attention path,
MLP path), each chosen with equal probability $\tfrac{1}{3}$.  The path distribution
has probability generating function $(1 + 2z)^L / 3^L$, mean $\mu = 2L/3$ in
block-level units ($\mu_{\text{fine}} = (7L+6)/3$), variance $\sigma^2 = 2L/9$,
and negative skewness $\gamma_1 = -(16/27)\,(9/(2L))^{3/2} < 0$.
\end{theorem}

\begin{proof}[Proof sketch]
At each parallel layer, there are three equally-likely branches, incrementing the
block-level path length by 0 (skip), 1 (attn), or 1 (mlp).  Thus
$X_i \in \{0, 1, 1\}$ with equal probability.  The pgf per layer is
$\tfrac{1}{3}(1 + z + z) = \tfrac{1}{3}(1 + 2z)$, giving $(1+2z)^L/3^L$ for $L$
layers.  The mean is $2L/3$ and the distribution is left-skewed because the mode
($\lfloor 2L/3 \rfloor$) exceeds the median for small $L$.  \qed
\end{proof}

\begin{theorem}[DenseNet Transformer: Super-Exponential Growth]
\label{thm:densenet}
A DenseNet transformer with $L$ layers has path count growing as
$|\pathcount| \sim 2^{\Omega(L \log L)}$.  The path distribution is left-skewed with
heavier short-path tails than the standard transformer.
\end{theorem}

\section{Empirical Path Distributions via Attribution Patching}
\label{sec:empirical}
% ============================================================================

The analytical distribution $\pathdist$ characterises the computational diversity an
architecture \emph{can} express.  A trained model, however, utilises different
subsets of its paths depending on the input and task.  We introduce the
\emph{empirical} path distribution $\emppathdist$ by applying Attribution Patching
to identify task-relevant edges, then running Algorithm~\ref{alg:pathcount} on the
resulting active subgraph.

\subsection{Attribution Patching for Edge Scoring}

\citet{meng2022locating} showed that activation patching can localise causal
contributions of individual model components. \citet{nanda2023attribution}
introduced Attribution Patching (AtP) as a first-order linear approximation:

\begin{definition}[AtP Score]
For a compute edge $e$ with activation $a_e$, the \emph{Attribution Patching score}
with respect to target logit $\mathcal{L}$ is:
$$\mathrm{AtP}(e) = \bigl\langle \nabla_{a_e}\,\mathcal{L}\big|_{\mathbf{a}^{\mathrm{base}}},\;
a_e^{\mathrm{clean}} - a_e^{\mathrm{base}} \bigr\rangle$$
where $\mathbf{a}^{\mathrm{clean}}$ are activations on the target input and
$\mathbf{a}^{\mathrm{base}}$ are activations on a baseline input.
\end{definition}

With a zero baseline ($\mathbf{a}^{\mathrm{base}} = \mathbf{0}$), this simplifies to
the gradient--activation product $\nabla_{a_e}\mathcal{L} \cdot a_e^{\mathrm{clean}}$,
a well-known attribution estimator closely related to integrated gradients
\citep{sundararajan2017axiomatic}.

\paragraph{Implementation.}
We anchor the computation graph at the first residual-stream checkpoint
(\texttt{blocks.0.hook\_resid\_pre}) via a detach--reattach hook (creating a leaf
tensor with \texttt{requires\_grad=True}).  Forward hooks on every
\texttt{hook\_attn\_out} and \texttt{hook\_mlp\_out} call \texttt{retain\_grad()},
preserving the intermediate gradients through a single backward pass from the target
logit. This avoids backpropagation through (potentially quantised) embedding weights
while retaining correct gradients for all transformer blocks.

For a given task, we compute AtP scores on $n$ input samples and pool them to obtain
a task-level attribution: $S_l^{\mathrm{attn}} = \frac{1}{n}\sum_{i}\mathrm{AtP}(e_l^{\mathrm{attn}},
x_i)$, and similarly $S_l^{\mathrm{mlp}}$.

\subsection{Active Subgraph and Empirical Distribution}

\begin{definition}[Active Subgraph]
\label{def:active}
Given threshold $\epsilon > 0$, the \emph{active subgraph}
$G_{\mathrm{active}} = (V, E_{\mathrm{active}})$ contains all skip edges (structurally
always active) and all compute edges $e$ with $\mathrm{AtP}(e) > \epsilon$:
$$E_{\mathrm{active}} = E_{\mathrm{skip}}\;\cup\;\bigl\{e \in E_{\mathrm{compute}} :
\mathrm{AtP}(e) > \epsilon\bigr\}.$$
\end{definition}

We set $\epsilon$ at the $q$-th quantile (default $q = 0.25$) of all AtP scores
pooled across a task's sample set, making it task-specific and
scale-invariant.\footnote{This choice keeps approximately $1 - q = 75\%$ of compute
edges active, providing a stable operating point. Sensitivity to $q$ is discussed in
Section~\ref{sec:experiments}.}

\begin{definition}[Empirical Path Distribution]
\label{def:empirical}
The \emph{empirical path distribution} $\emppathdist$ is obtained by running
Algorithm~\ref{alg:pathcount} on $G_{\mathrm{active}}$:
$$\emppathdist(\ell) = \frac{|\{p \in \pathcount(G_{\mathrm{active}}) : |p| = \ell\}|}{%
|\pathcount(G_{\mathrm{active}})|}$$
\end{definition}

Since $G_{\mathrm{active}} \subseteq G$, the support of $\emppathdist$ is a
subset of the support of $\pathdist$.  Removing an active compute edge from the graph
collapses all paths that passed through it onto the corresponding skip paths, shifting
mass toward shorter lengths.

\subsection{Synergy Gap}
\label{sec:synergy}

\begin{definition}[Synergy Gap]
\label{def:synergy}
The \emph{Synergy Gap} between the analytical and empirical path distributions is:
$$\gap = \entropy(\pathdist) - \entropy(\emppathdist) \geq 0$$
\end{definition}

The inequality holds because $G_{\mathrm{active}} \subseteq G$ reduces distributional
spread.  Intuitively:
\begin{itemize}
    \item $\gap \approx 0$: the model uses essentially all computational paths, close
          to the architectural maximum---consistent with simple, low-complexity tasks
          where many routes reach the right answer.
    \item $\gap \gg 0$: the model concentrates on a narrow subset of paths, indicating
          strong specialisation.  We predict $\gap$ grows with task complexity
          (Section~\ref{sec:experiments}).
\end{itemize}

The Synergy Gap complements the analytical entropy: a model with high $\entropy(\pathdist)$
but also high $\gap$ has rich architectural potential that it selectively exploits for
complex tasks.

\section{Path-Derived Metrics for Emergence}
\label{sec:metrics}
% ============================================================================

We define four metrics derived from the analytical and empirical path distributions.

\subsection{Path Entropy}

\begin{definition}[Path Entropy]
$$\entropy(\pathdist) = -\sum_{\ell} \pathdist(\ell) \log_2 \pathdist(\ell)$$
\end{definition}

Path entropy measures the \emph{diversity} of computational depths available in the
architecture.  For the standard pre-LN transformer in block-level units,
$\entropy(\pathdist) = \frac{1}{2}\log_2(\pi e L) + O(1/L) \approx \frac{1}{2}\log_2 L + C$
(by Stirling approximation of Binomial$(2L, \tfrac{1}{2})$).

\subsection{Effective Depth Spectrum Width}

\begin{definition}[Effective Depth Spectrum Width]
$$\hat{\sigma}_d = \frac{\ell_{\max} - \ell_{\min}}{\ell_{\max}}$$
\end{definition}

This captures the range of computational depths available.  Architectures with
cross-layer residuals have $\hat{\sigma}_d$ close to 1; Highway transformers have
smaller $\hat{\sigma}_d$.

\subsection{Tail-Mass Ratio}

\begin{definition}[Tail-Mass Ratio]
For integer cutoff $k$ (default $k = \lfloor\E_{\pathdist}[L]\rfloor$):
$$\tau_k = \frac{\Pr_{\pathdist}(L > k)}{\Pr_{\pathdist}(L \leq k)}$$
\end{definition}

The tail-mass ratio captures the balance between deep (long-path) and shallow
(short-path) computation.  For symmetric distributions (standard transformer),
$\tau_k = 1$ at $k = \mu$.  Left-skewed architectures have $\tau_k > 1$, indicating
heavier long-path mass.  We define $\hat{\tau}_k$ analogously for $\emppathdist$.

\subsection{Synergy Gap (Empirical Metric)}

As defined in Section~\ref{sec:synergy}:
$$\gap = \entropy(\pathdist) - \entropy(\emppathdist) \geq 0$$

This is the only metric that requires a trained model and task inputs; the others are
purely architectural.

\paragraph{Emergence hypothesis.}
We conjecture that emergence occurs when the analytical path entropy
$\entropy(\pathdist)$ exceeds a task-dependent threshold $\entropy^*(T)$:
\begin{equation}
\label{eq:emergence}
\text{Task } T \text{ emerges at depth } L^* \text{ where }
\entropy(\pathdist_{L^*}) \geq \entropy^*(T).
\end{equation}
Concurrently, emergence may coincide with the \emph{Synergy Gap decreasing}:
as a model becomes large enough to express the required computation, it recruits
longer paths, reducing $\gap$ even as task performance improves.

\section{Experimental Design}
\label{sec:experiments}
% ============================================================================

We structure experiments to test four hypotheses derived from the theory.

\subsection{Experimental Setup}

\paragraph{Architectures.} We train models from scratch across 8 architectural
families: (1)~LLaMA-style pre-RMSNorm, (2)~GPT-2-style pre-LN, (3)~Post-LN,
(4)~GPT-J parallel, (5)~Highway, (6)~DenseNet transformer, (7)~KIMI skip-2 residual,
(8)~U-Net transformer.  For each family we train at 5 scale points:
125M, 350M, 1.3B, 6.7B, and 13B parameters.

\paragraph{Training.} All models are trained on the same data mixture (300B tokens
from RedPajama-v2) with identical hyperparameters per scale point following
\citet{hoffmann2022training}.  The only variation is the residual connection topology.

\paragraph{Evaluation tasks (12 total).} Tasks are ordered by compositional
complexity:
\begin{itemize}
    \item \textbf{Low}: sentiment (SST-2), topic classification (AG News), NLI (RTE).
    \item \textbf{Medium}: reading comprehension (BoolQ), commonsense reasoning
          (HellaSwag, WinoGrande), word-in-context (WiC).
    \item \textbf{High}: multi-hop reasoning (HotpotQA), arithmetic (GSM8K), code
          generation (HumanEval), chain-of-thought math (MATH), abstract reasoning
          (ARC-Challenge).
\end{itemize}

For preliminary empirical path distribution experiments (Section~\ref{sec:empirical}),
we evaluate on three representative tasks---SST-2 (Low), BoolQ (Medium), GSM8K
(High)---using pre-trained models loaded via TransformerLens
\citep{nanda2022transformerlens}.  The full 12-task evaluation requires the large-scale
training runs described in Experiments 1--2.

\subsection{Experiment 1: Path Entropy Predicts Emergence Threshold}

\paragraph{Hypothesis.} For each task $T$, there exists a critical path entropy
$\entropy^*(T)$ such that performance transitions from random to above-chance when
$\entropy(\pathdist) \geq \entropy^*(T)$.

\paragraph{Method.} For each architecture $\times$ scale combination, we compute
$\entropy(\pathdist)$ analytically (depends only on architecture and depth, not
learned weights).  We plot task accuracy vs.\ $\entropy(\pathdist)$ and fit a sigmoid:
$$\mathrm{Acc}(\entropy) = \frac{a}{1 + \exp(-b(\entropy - \entropy^*))} + c$$
to estimate $\entropy^*$ for each task.  We compare sigmoid sharpness $b$ to fits
against $\log(\text{params})$ and $\log(\text{FLOPs})$.

\paragraph{Expected results.}
\begin{enumerate}
    \item Sigmoid fits against $\entropy(\pathdist)$ have higher $R^2$ than fits
          against model size.
    \item $\entropy^*$ increases monotonically with task compositional complexity.
    \item Architectures with higher path entropy (DenseNet, MoD) achieve emergence at
          smaller parameter counts.
\end{enumerate}

\subsection{Experiment 2: Architectural Ablation of Residual Topology}

\paragraph{Hypothesis.} Modifying the residual connection topology changes $\pathdist$
and consequently shifts the emergence threshold, independent of parameter count.

\paragraph{Method.} Starting from a 1.3B-parameter LLaMA baseline ($\entropy = 3.34$
bits at $L = 32$), we construct 6 variants differing only in residual topology:
\begin{enumerate}
    \item Baseline: standard pre-RMSNorm ($\entropy = 3.34$).
    \item +Skip-2: attention skip-2 connections ($\entropy = 3.60$).
    \item +DenseNet: every-layer-to-every-layer ($\entropy = 3.86$).
    \item +U-Net: symmetric encoder-decoder skips ($\entropy = 3.39$).
    \item $-$Half-residuals: remove MLP residuals ($\entropy = 2.84$).
    \item $-$All-residuals: remove all residuals ($\entropy = 0$, single path).
\end{enumerate}
All variants have identical parameter count and FLOPs.  We train each for 100B tokens
and evaluate on all 12 tasks.

\paragraph{Expected results.} DenseNet $>$ Skip-2 $>$ U-Net $\approx$ Baseline $>$
Half-residuals $\gg$ No-residuals on high-complexity tasks, with smaller differences on
low-complexity tasks.

\subsection{Experiment 3: Empirical Path Distribution across Tasks}

\paragraph{Hypothesis.} The empirical path entropy $\entropy(\emppathdist)$ decreases
relative to the analytical entropy as task complexity increases
(i.e.\ $\gap$ grows with complexity), reflecting increasing specialisation of the model's
computational paths for complex reasoning.

\paragraph{Method.} For each (model, task) pair, we:
\begin{enumerate}
    \item Run $n = 50$ sample prompts through the model with
          \texttt{PathAnalyzer.compute\_attribution\_scores()}, computing an AtP
          gradient--activation score for every attention and MLP edge per sample.
    \item Set a task-level threshold $\epsilon$ at the $25$th percentile of all AtP
          scores across the sample set.
    \item Apply the DP path counter (Algorithm~\ref{alg:pathcount}) to $G_{\mathrm{active}}$
          per sample; report mean and std of $\entropy(\emppathdist)$ across samples.
    \item Report the Synergy Gap $\gap = \entropy(\pathdist) - \E[\entropy(\emppathdist)]$.
\end{enumerate}

\paragraph{Expected results.}
\begin{enumerate}
    \item $\entropy(\emppathdist)$ is lower for GSM8K than BoolQ than SST-2 for all
          models.
    \item $\gap$ increases monotonically with task complexity.
    \item Larger models (more parameters) show smaller $\gap$ on the same task,
          consistent with the hypothesis that scaling enables using more of the
          available computational paths.
\end{enumerate}

Figure~\ref{fig:entropy_complexity} shows this experiment applied to pre-trained
models (Llama-3-8B, GPT-J-6B, Falcon-7B) using the \texttt{experiment\_runner.py}
pipeline.

\subsection{Experiment 4: Cross-Architecture Scaling Law}

\paragraph{Hypothesis.} The scaling-law exponent $\alpha$ in
$\mathrm{Loss} \sim C^{-\alpha}$ is predicted by path distribution properties.

\paragraph{Method.} For each of the 8 architectures across 5 scale points, we fit
$\mathrm{Loss}(C) = aC^{-\alpha} + b$ and regress the fitted exponent $\alpha$ against
$\entropy(\pathdist)$, $\sigma_d$, $\tau_k$, and $\log_2 |\pathcount|$.

\paragraph{Expected results.} $\alpha$ positively correlates with path entropy:
architectures with richer path distributions are more compute-efficient per FLOP.

\section{Architectural Design Principles}
\label{sec:design}
% ============================================================================

Path distribution theory suggests concrete design principles for architectures
optimised for emergent abilities:

\paragraph{Principle 1: Maximise path entropy.} Add cross-layer residual connections
(DenseNet or skip-$k$ style) to increase $\entropy(\pathdist)$ without increasing
parameter count.  Our theory predicts this is more effective than simply adding layers.

\paragraph{Principle 2: Control skewness.} Left-skewed distributions provide more
short paths (robust simple-task performance) while still maintaining a heavy long-path
tail for complex reasoning.

\paragraph{Principle 3: Positive excess kurtosis.} U-Net and MoD architectures exhibit
leptokurtic distributions (heavier tails than binomial), providing extreme short and
long paths that may be critical for sharp emergence transitions.

\paragraph{Principle 4: Monitor Synergy Gap.} Architectures that produce large $\gap$
for high-complexity tasks are more interpretable (concentrated active paths) but may
be underutilising structural capacity.  Conversely, small $\gap$ on simple tasks
suggests the model wastes compute on diversity where it is not needed.

\paragraph{Proposed architecture: PathNet.} We propose a hybrid architecture combining
local intra-layer residuals (LLaMA-style) with logarithmically-spaced cross-layer
residuals: layer $i$ connects to layers $i + 2^j$ for
$j = 1, 2, \ldots, \lfloor\log_2(L - i)\rfloor$.  This yields:
\begin{itemize}
    \item Path count growing as $2^{\Theta(L \log L)}$, faster than standard
          transformers but manageable.
    \item A path distribution with controlled negative skewness and high entropy.
    \item No additional parameters (residual additions are parameter-free).
\end{itemize}

\section{Theoretical Analysis}
\label{sec:analysis}
% ============================================================================

\subsection{Why Short Paths and Long Paths Both Matter}

Consider the decomposition of a model's output into contributions from paths.  Each
path $p$ computes a function $f_p(\bx)$, and the model's output is:
$$f(\bx; \btheta) = \sum_{p \in \pathcount} f_p(\bx; \btheta_p)$$

\textbf{Short paths} (few compute blocks) compute simple functions: they are easy to
learn and generalise well on simple tasks---analogous to the ``shallow subnetworks''
of \citet{veit2016residual}.

\textbf{Long paths} (many blocks) compose many nonlinear operations, enabling complex
multi-step reasoning.  They are harder to train (vanishing gradients, longer credit
assignment) and require more data.

\textbf{The emergence mechanism.} At a critical model size, long paths accumulate
sufficient gradient signal to learn compositional functions.  Before this threshold,
only short paths are trained and the Synergy Gap is high; after it, both short and
long paths contribute and $\gap$ decreases.  The analytical path entropy determines
how many long paths exist (the ``computational surface area'' available), while the
empirical entropy measures which portion of that surface is actually utilised.

\subsection{Early vs.\ Late Layer Residuals}

\begin{proposition}[Layer Position Effect]
In a standard pre-LN transformer with $L$ layers, a residual connection at layer $i$
participates in $4^{L-i-1}$ paths through the network.  Therefore, early-layer
residuals contribute exponentially more paths than late-layer residuals.
\end{proposition}

This explains why DenseNet transformers, which add connections from \emph{every}
early layer to \emph{every} late layer, achieve the fastest path-count growth.

\subsection{Connection to Circuit Complexity}

Each path of block-level length $\ell$ through the transformer corresponds to a
circuit of depth $\ell$.  The path distribution therefore defines a
\emph{mixture of circuits at different depths}: a model with path distribution
$\pathdist$ can express any function computable by a weighted mixture of circuits
whose depth distribution matches $\pathdist$.  The empirical path distribution
$\emppathdist$ identifies which circuits in this mixture are actually recruited for a
given task, connecting the distributional framework to the circuit-finding literature
\citep{conmy2023automated}.


% ============================================================================
\section{Discussion and Limitations}
\label{sec:discussion}
% ============================================================================

\paragraph{Limitations.}
(1)~Path analysis treats all paths as equally likely a priori; the empirical
distribution (Section~\ref{sec:empirical}) partially addresses this by reflecting
learned weights, but uses a first-order AtP approximation that may miss nonlinear
interactions between paths.
(2)~The block-level representation in the code uses $\delta \in \{0, 1\}$ and
fine-grained theory uses $\delta \in \{1, 3\}$; absolute path-length values are
not directly comparable across representations (shape statistics are preserved).
(3)~The large-scale experiments (Experiments 1--2) require approximately 335{,}000
A100-hours (Appendix~\ref{app:compute}) and results are currently presented as
predictions.
(4)~The AtP threshold $\epsilon$ introduces a degree of freedom; we use a
data-driven quantile approach but the sensitivity of $\gap$ to this choice merits
further study.
(5)~Our theory does not yet account for attention-specific effects: different
attention heads may gate different residual contributions, creating effectively
richer path topologies than the block-level DAG represents.

\paragraph{Broader impact.} Path distribution theory provides a principled way to
design architectures for specific computational profiles, potentially reducing the
cost of achieving emergent abilities.  However, making powerful AI capabilities
accessible at smaller scales raises dual-use concerns.


% ============================================================================
\section{Conclusion}
\label{sec:conclusion}
% ============================================================================

We have introduced path distribution theory as a framework for understanding emergent
abilities in transformers.  By characterising the exact path distributions of 8
architectural families, we showed that the combinatorial structure of residual
connections---captured by path entropy, depth spectrum width, tail-mass ratio, and the
novel Synergy Gap---provides a rich set of metrics for both architectural design and
interpretability.

A key contribution is the empirical path distribution pipeline: by applying
Attribution Patching to score computational edges and restricting the path-counting DP
to the resulting active subgraph, we obtain a task-specific measure of how a trained
model utilises its theoretical computational capacity.  The Synergy Gap between
analytical and empirical entropies serves as a diagnostic of path specialisation that
is distinct from accuracy, probing model behaviour at the level of computational
routing rather than prediction correctness.

Our experimental designs, implemented in open-source code (\texttt{path\_analyzer.py},
\texttt{experiment\_runner.py}), are falsifiable and reproducible at small scale using
pre-trained models and standard benchmarks.  We hypothesise that emergence correlates
with a critical path entropy threshold ($\entropy \approx 3.2$ bits at 32 layers) and
that the Synergy Gap grows monotonically with task complexity across model families.
Confirming or refuting these predictions at scale is the central challenge left for
future work.

\section{Algorithms}
\label{app:algorithm}
% ============================================================================

\begin{algorithm}[H]
\SetAlgoLined
\KwIn{DAG $G=(V,E)$ where each edge $e=(u,v)$ carries $\delta_e \in \{0,1\}$}
\KwOut{Path distribution $\pathdist(\ell)$ for $\ell = 0,\ldots,\ell_{\max}$}
Initialise $\mathrm{dp}[v] \gets \mathbf{0}$ for all $v \in V$\;
$\mathrm{dp}[\mathrm{source}][0] \gets 1$\tcp*{one path of length 0 at source}
\ForEach{$v \in \mathrm{TopologicalSort}(G)$}{
    \ForEach{edge $(v, u, \delta_e) \in E$}{
        \ForEach{$\ell = 0, \ldots, \ell_{\max} - \delta_e$}{
            $\mathrm{dp}[u][\ell + \delta_e] \mathrel{+}= \mathrm{dp}[v][\ell]$\;
        }
    }
}
\Return{$\mathrm{dp}[\mathrm{sink}] \;/\; \|\mathrm{dp}[\mathrm{sink}]\|_1$}\;
\caption{Dynamic-programming path distribution. Skip edges have $\delta_e=0$;
compute edges have $\delta_e=1$. Complexity: $O(|E| \cdot \ell_{\max})$.}
\label{alg:pathcount}
\end{algorithm}

\vspace{1em}
\noindent\textbf{Correctness note.}  A common implementation error is to increment
$\ell$ by 1 for \emph{every} edge.  Since the DAG has \emph{two} edges per binary
branch point (one skip, one compute, both connecting the same pair of nodes), this
collapses all paths to the same length and produces a degenerate point mass.  The
correct implementation uses edge-specific $\delta_e$ as shown above, which is what
\texttt{PathAnalyzer.\_path\_count\_dp()} implements.

\begin{algorithm}[H]
\SetAlgoLined
\KwIn{Tokens $x$; trained model $\theta$; quantile $q \in (0,1)$; $n$ task samples}
\KwOut{Empirical distribution $\emppathdist$; Synergy Gap $\gap$}
Compute $\pathdist$ analytically via Algorithm~\ref{alg:pathcount} on full $G$\;
\For{$i = 1, \ldots, n$}{
    Forward pass on $x_i$, retaining $\{a_l^{\text{attn}}, a_l^{\text{mlp}}\}_{l=1}^L$\;
    Backward pass from target logit, reading $\{\nabla a_l^{\text{attn}},\nabla a_l^{\text{mlp}}\}_{l=1}^L$\;
    $S_l^{\text{attn},i} \gets |\nabla a_l^{\text{attn}} \odot a_l^{\text{attn}}|_{\text{mean}}$\;
    $S_l^{\text{mlp},i}  \gets |\nabla a_l^{\text{mlp}}  \odot a_l^{\text{mlp}} |_{\text{mean}}$\;
}
$\epsilon \gets \mathrm{quantile}_q\bigl(\{S_l^{\text{attn},i}\} \cup \{S_l^{\text{mlp},i}\}\bigr)$\;
\For{$i = 1, \ldots, n$}{
    $E_{\text{active}}^i \gets E_{\text{skip}} \cup \{e : S_e^i > \epsilon\}$\;
    $\emppathdist_i \gets$ Algorithm~\ref{alg:pathcount} on $G_{\text{active}}^i$\;
}
$\emppathdist \gets \frac{1}{n}\sum_i \emppathdist_i$;\ \
$\gap \gets \entropy(\pathdist) - \entropy(\emppathdist)$\;
\Return{$\emppathdist,\; \gap$}\;
\caption{Empirical path distribution via Attribution Patching.}
\label{alg:atp}
\end{algorithm}
